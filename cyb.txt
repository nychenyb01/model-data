        document.addEventListener('DOMContentLoaded', function() {
            // 初始化图表
            const ctx = document.getElementById('paymentChart').getContext('2d');
            const paymentChart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: ['原总利息', '新总利息', '原还款总额', '新还款总额'],
                    datasets: [{
                        label: '金额（元）',
                        data: [2240650.80, 2240650.80, 6650650.80, 6650650.80],
                        backgroundColor: [
                            'rgba(54, 162, 235, 0.5)',
                            'rgba(75, 192, 192, 0.5)',

                        ],
                        borderColor: [
                            'rgb(54, 162, 235)',
                            'rgb(75, 192, 192)',
                            'rgb(255, 99, 132)',
                            'rgb(255, 159, 64)'
                        ],
                        borderWidth: 1
                    }]
                },
                options: {
                    responsive: true,
                    plugins: {
                        title: {
                            display: true,
                            text: '还款对比分析'
                        },
                    },
                    scales: {
                        y: {
                            beginAtZero: true,
                            ticks: {
                                callback: function(value) {
                                    return value.toLocaleString() + '元';
                                }
                            }
                        }
                    }
                }
            });
            
            // 初始化还款计划表
            updatePaymentTable();
            
            // 计算按钮点击事件
            document.getElementById('calculateBtn').addEventListener('click', function() {
                calculateLoan();
            });
            
            // 初始化计算
            calculateLoan();
            
            // 计算贷款函数
            function calculateLoan() {
                // 获取输入值
                const loanAmount = parseFloat(document.getElementById('loanAmount').value) * 10000;
                const loanTerm = parseInt(document.getElementById('loanTerm').value);
                const annualRate = parseFloat(document.getElementById('annualRate').value) / 100;
                const currentMonth = parseInt(document.getElementById('currentMonth').value);
                const prepayAmount = parseFloat(document.getElementById('prepayAmount').value) * 10000;
                const prepayMonth = parseInt(document.getElementById('prepayMonth').value);
                const prepayTimes = parseInt(document.getElementById('prepayTimes').value);
                
                // 计算月利率
                const monthlyRate = annualRate / 12;
                
                // 计算总期数
                const totalMonths = loanTerm * 12;
                
                // 计算原月供
                const monthlyPayment = loanAmount * monthlyRate * Math.pow(1 + monthlyRate, totalMonths) / 
                                      (Math.pow(1 + monthlyRate, totalMonths) - 1);
                
                // 计算原总利息
                const totalInterest = monthlyPayment * totalMonths - loanAmount;
                
                // 计算剩余本金（等额本息法）
                let remainingPrincipal = loanAmount * (Math.pow(1 + monthlyRate, totalMonths) - Math.pow(1 + monthlyRate, currentMonth - 1)) / 
                                        (Math.pow(1 + monthlyRate, totalMonths) - 1);
                
                // 计算提前还款后的新贷款情况
                let newRemainingPrincipal = remainingPrincipal;
                let totalPaidInterest = 0;
                let newTotalMonths = 0;
                let prepaid = 0;
                
                // 模拟还款过程
                for (let month = currentMonth; month <= totalMonths; month++) {
                    if (newRemainingPrincipal <= 0) break;
                    
                    // 当月利息
                    const monthlyInterest = newRemainingPrincipal * monthlyRate;
                    totalPaidInterest += monthlyInterest;
                    
                    // 当月本金
                    const monthlyPrincipal = monthlyPayment - monthlyInterest;
                    
                    // 提前还款处理
                    let extraPayment = 0;
                    if (month >= prepayMonth && month < prepayMonth + prepayTimes && prepaid < prepayAmount) {
                        extraPayment = Math.min(prepayAmount / prepayTimes, newRemainingPrincipal - monthlyPrincipal);
                        prepaid += extraPayment;
                    }
                    
                    // 更新剩余本金
                    newRemainingPrincipal -= (monthlyPrincipal + extraPayment);
                    
                    // 计数
                    newTotalMonths++;
                    
                    // 如果剩余本金非常小，直接清零
                    if (newRemainingPrincipal < 0.01) {
                        newRemainingPrincipal = 0;
                    }
                }
                
                // 计算新总利息
                const newTotalInterest = totalPaidInterest;
                
                // 计算节省的利息
                const savedInterest = totalInterest - newTotalInterest;
                
                // 计算新还款总额
                const newTotalPayment = remainingPrincipal + newTotalInterest;
                
                // 更新页面显示
                document.getElementById('originalMonthly').textContent = monthlyPayment.toFixed(2) + '元';
                document.getElementById('originalInterest').textContent = totalInterest.toFixed(2) + '元';
                document.getElementById('remainingPrincipal').textContent = remainingPrincipal.toFixed(2) + '元';
                document.getElementById('newMonthly').textContent = monthlyPayment.toFixed(2) + '元';
                document.getElementById('remainingInterest').textContent = newTotalInterest.toFixed(2) + '元';
                document.getElementById('savedInterest').textContent = savedInterest.toFixed(2) + '元';
                document.getElementById('remainingMonths').textContent = newTotalMonths + '个月';
                
                document.getElementById('originalTotal').textContent = (loanAmount + totalInterest).toFixed(2) + '元';
                document.getElementById('newTotal').textContent = newTotalPayment.toFixed(2) + '元';
                document.getElementById('totalSavings').textContent = (loanAmount + totalInterest - newTotalPayment).toFixed(2) + '元';
                document.getElementById('prepayPercentage').textContent = ((prepayAmount / loanAmount) * 100).toFixed(2) + '%';
                
                document.getElementById('savingsAmount').textContent = savedInterest.toFixed(2) + '元';
                document.getElementById('reducedMonths').textContent = (totalMonths - currentMonth + 1 - newTotalMonths) + '个月';
                
                // 更新图表
                paymentChart.data.datasets[0].data = [
                    totalInterest, 
                    newTotalInterest, 
                    loanAmount + totalInterest, 
                    newTotalPayment
                ];
                paymentChart.update();
                
                // 更新还款计划表
                updatePaymentTable();
            }
            
            // 更新还款计划表
            function updatePaymentTable() {
                const tableBody = document.getElementById('paymentTableBody');
                tableBody.innerHTML = '';
                
                const loanAmount = parseFloat(document.getElementById('loanAmount').value) * 10000;
                const annualRate = parseFloat(document.getElementById('annualRate').value) / 100;
                const monthlyRate = annualRate / 12;
                const totalMonths = parseInt(document.getElementById('loanTerm').value) * 12;
                const monthlyPayment = loanAmount * monthlyRate * Math.pow(1 + monthlyRate, totalMonths) / 
                                      (Math.pow(1 + monthlyRate, totalMonths) - 1);
                
                const prepayAmount = parseFloat(document.getElementById('prepayAmount').value) * 10000;
                const prepayMonth = parseInt(document.getElementById('prepayMonth').value);
                const prepayTimes = parseInt(document.getElementById('prepayTimes').value);
                
                let remainingPrincipal = loanAmount;
                let prepaid = 0;
                
                // 只显示最近12个月
                const displayMonths = Math.min(12, totalMonths);
                
                for (let month = 1; month <= displayMonths; month++) {
                    const row = document.createElement('tr');
                    
                    // 期数
                    const monthCell = document.createElement('td');
                    monthCell.textContent = month;
                    row.appendChild(monthCell);
                    
                    // 还款日期（模拟）
                    const dateCell = document.createElement('td');
                    const currentDate = new Date();
                    currentDate.setMonth(currentDate.getMonth() + month - 1);
                    dateCell.textContent = currentDate.toLocaleDateString();
                    row.appendChild(dateCell);
                    
                    // 月供本金
                    const monthlyInterest = remainingPrincipal * monthlyRate;
                    const monthlyPrincipal = monthlyPayment - monthlyInterest;
                    const principalCell = document.createElement('td');
                    principalCell.textContent = monthlyPrincipal.toFixed(2);
                    row.appendChild(principalCell);
                    
                    // 月供利息
                    const interestCell = document.createElement('td');
                    interestCell.textContent = monthlyInterest.toFixed(2);
                    row.appendChild(interestCell);
                    
                    // 提前还款
                    let extraPayment = 0;
                    if (month >= prepayMonth && month < prepayMonth + prepayTimes && prepaid < prepayAmount) {
                        extraPayment = Math.min(prepayAmount / prepayTimes, remainingPrincipal - monthlyPrincipal);
                        prepaid += extraPayment;
                    }
                    const prepayCell = document.createElement('td');
                    prepayCell.textContent = extraPayment > 0 ? extraPayment.toFixed(2) : '-';
                    row.appendChild(prepayCell);
                    
                    // 剩余本金
                    remainingPrincipal -= (monthlyPrincipal + extraPayment);
                    const remainingCell = document.createElement('td');
                    remainingCell.textContent = remainingPrincipal.toFixed(2);
                    row.appendChild(remainingCell);
                    
                    tableBody.appendChild(row);
                }
            }
        });
